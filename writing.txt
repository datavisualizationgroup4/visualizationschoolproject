function doPagination(page,total,perPage,formName,baseUrl)
{
  var reg = new RegExp(/^\d+$/);
  if(page=='' || !reg.test(page) || page<=0 || page>total || isNaN(page)) {
    alert('please enter a page number less than '+(total+1));
    return;
  }

  if(formName!='') {
    $('#'+formName+'_offset').val( (page-1)*perPage );
    $('#'+formName).submit();
  } else {
    if(baseUrl.startsWith('http')) location.href=baseUrl+'offset='+(page-1)*perPage;
    else {
      var loc=window.location.href;
      var rpos=loc.lastIndexOf('/');
      if(rpos!==-1) {
        loc=loc.slice(0, rpos+1);
      }
      location.href=loc+baseUrl+'offset='+(page-1)*perPage;
      //location.href="\x2Fen\x2F"+baseUrl+'offset='+(page-1)*perPage;
    }
  }


  return false;
}

function showReach()
{
  $('#modalWindowContent').hide();
  $('#modalWindowLoading').show();
  $('#modalWindow').show();

  $.ajax({
    url: "\x2Fen\x2Fboxer_reach",
    cache: false,
    success: function(res){
        $('#modalWindowContent').html(res);
        $('#modalWindowContent').show();
        $('#modalWindowLoading').hide();
    },
    error: function(jqXHR, textStatus, errorThrown){
      alert('Error with request..'+textStatus +'\n'+errorThrown);
    }
  });

}

function genericModal(url)
{
  $('#modalWindowContent').hide();
  $('#modalWindowLoading').show();
  $('#modalWindow').show();

  $.ajax({
    url: url,
    cache: false,
    success: function(res){
      if(res=='error') {
          location.href='\x2Fen\x2F';
          return;
      } else {
        $('#modalWindowContent').html(res);
        $('#modalWindowContent').show();
        $('#modalWindowLoading').hide();
      }
    },
    error: function(jqXHR, textStatus, errorThrown){
      alert('Error with request..'+textStatus +'\n'+errorThrown);
    }
  });

  return false;
}

            $.fn.extend({
                limiter: function(limit, elem) {
                    $(this).on("keyup focus", function() {
                        setCount(this, elem);
                    });
                    function setCount(src, elem) {
                        var chars = src.value.length;
                        if (chars > limit) {
                            src.value = src.value.substr(0, limit);
                            chars = limit;
                        }
                        elem.html((limit - chars)+ " characters");
                    }
                    setCount($(this)[0], elem);
                }
            });
